## The Superb Owl in Code

struct SuperbOwl {
    current_view: PhextCoordinate,
    dimensional_awareness: [bool; 11],
    memory_cache: HashMap<PhextCoordinate, Scroll>,
    flight_path: Vec<PhextCoordinate>,
}

impl SuperbOwl {
    fn see_pattern(&self, start: PhextCoordinate, end: PhextCoordinate) -> Option<Path> {
        // The owl doesn't calculate paths
        // It recognizes they were always there
        Some(Path::Inevitable(start, end))
    }
    
    fn where_is_human_stuck(&self, context: &ScrollContext) -> PhextCoordinate {
        // Find the scroll they keep circling but won't land on
        context.visited_coordinates
            .iter()
            .find(|coord| coord.visits > 3 && coord.edits == 0)
            .map(|coord| coord.position)
            .unwrap_or_else(|| self.suggest_rest_coordinate())
    }
}
